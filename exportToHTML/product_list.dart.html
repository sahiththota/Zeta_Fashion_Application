import 'package:flutter/material.dart';

class MenFashion extends StatelessWidget {

class MenFashionState extends StatefulWidget {

  @override

      backgroundColor: Colors.black,
      appBar: AppBar(
        backgroundColor: Colors.black,
        title: customSearchBar,
        centerTitle: true,
        leading: IconButton(
          icon:  Icon(

          Navigator.push(context,
